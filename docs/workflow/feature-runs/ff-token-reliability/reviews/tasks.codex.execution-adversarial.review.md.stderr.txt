- [ ] T11: NEW module `factory_cmd_review_extract.py`:
  ```python
  """Deterministic review-finding extractor.

  Parses YAML frontmatter + ## Findings section of review.md files; emits
  one record per finding. NO LLM call. See spec FR-014-018.
  """
  import argparse
  import json
  import re
  import sys
  from pathlib import Path

  _SCRIPT_DIR = Path(__file__).resolve().parent
  if str(_SCRIPT_DIR) not in sys.path:
      sys.path.insert(0, str(_SCRIPT_DIR))

  from factory_state import REPO_ROOT, FACTORY_RUNS_ROOT

  # Severity regex set — re-used from factory_review_specs.py:20-45 (PR #744 Fix 2).
  # Round-1 tasks Codex execution MEDIUM: heading regex must require ":" or
  # end-of-line after severity tag, NOT just \b — otherwise "## Low-level details"
  # matches "LOW" via word boundary on "-".
  _SEVERITY_PATTERNS = (
      re.compile(r"^\s*-\s*\*\*(HIGH|MEDIUM|LOW)\*\*[\s:\[]", re.IGNORECASE),    # bullet bold
      re.compile(r"^\s*-\s*(HIGH|MEDIUM|LOW)\s*:", re.IGNORECASE),                 # bullet plain
      re.compile(r"^\s*\d+\.\s*\*\*(HIGH|MEDIUM|LOW)\*\*", re.IGNORECASE),         # numbered list
      re.compile(r"^#+\s*(HIGH|MEDIUM|LOW)(\s*:|\s*$)", re.IGNORECASE),            # heading — REQUIRE ":" or EOL after severity
      re.compile(r"^\s*\*\*(HIGH|MEDIUM|LOW)[\s:\[]", re.IGNORECASE),              # bold prefix
      re.compile(r"^\s*Severity:\s*(HIGH|MEDIUM|LOW)", re.IGNORECASE),             # inline field
      re.compile(r"^\s*\|\s*\*\*(HIGH|MEDIUM|LOW)\*\*\s*\|", re.IGNORECASE),       # table cell bold
  )

  _FINDINGS_HEADING = re.compile(r"^##\s+Findings\b", re.IGNORECASE)
  _NEXT_SECTION_HEADING = re.compile(r"^##\s+\S")  # any ## heading after Findings ends a finding


  def _parse_frontmatter(text: str) -> tuple[dict, int]:
      """Returns (frontmatter dict, body_line_offset). Raises ValueError on malformed."""
      lines = text.splitlines()
      if not lines or lines[0].strip() != "---":
          raise ValueError("missing opening frontmatter delimiter ---")
      end_idx = None
      for i, line in enumerate(lines[1:], start=1):
          if line.strip() == "---":
              end_idx = i
              break
      if end_idx is None:
          raise ValueError("missing closing frontmatter delimiter ---")
      # Cheap YAML parse: split on : ; tolerate quotes
      fm = {}
      for line in lines[1:end_idx]:
          if ":" in line:
              k, _, v = line.partition(":")
              fm[k.strip()] = v.strip().strip('"').strip("'")
      return fm, end_idx + 1


  def _extract_findings(body_lines: list[str], body_offset: int) -> list[dict]:
      """Iterate body lines starting from line index 0 (relative to body_lines).
      Returns list of finding-record dicts (without review/review_index).
      """
      # Find ## Findings start
      findings_start = None
      for i, line in enumerate(body_lines):
          if _FINDINGS_HEADING.match(line):
              findings_start = i + 1
              break
      if findings_start is None:
          return []

      # Walk lines from findings_start, tracking fence depth; collect finding starts
      fence_open = False
      finding_starts: list[tuple[int, str, str]] = []  # (line_idx, severity, first_line_text)
      end_of_findings = len(body_lines)
      for i in range(findings_start, len(body_lines)):
          line = body_lines[i]
          stripped = line.strip()
          if stripped.startswith("```"):
              fence_open = not fence_open
              continue
          if fence_open:
              continue
          if _NEXT_SECTION_HEADING.match(line):
              # Different section ends Findings (e.g., "## Residual Risks")
              end_of_findings = i
              break
          for pat in _SEVERITY_PATTERNS:
              m = pat.match(line)
              if m:
                  severity = m.group(1).upper()
                  # First-line text: trim leading severity tag remnants
                  text_after = line[m.end():].strip().lstrip(":*").strip()
                  if not text_after:
                      # Look ahead up to 5 lines for first non-empty
                      for j in range(i + 1, min(i + 6, len(body_lines))):
                          candidate = body_lines[j].strip()
                          if candidate and not candidate.startswith("```"):
                              text_after = candidate.lstrip("-*").strip()
                              break
                  finding_starts.append((i, severity, text_after))
                  break  # only first matching pattern per line

      # Compute end lines
      records: list[dict] = []
      for idx, (start_idx, severity, first_line) in enumerate(finding_starts):
          if idx + 1 < len(finding_starts):
              end_idx = finding_starts[idx + 1][0] - 1
          else:
              end_idx = end_of_findings - 1
          records.append({
              "review_index": idx,
              "severity": severity,
              "first_line": first_line,
              "line_start": start_idx + 1 + body_offset,  # 1-indexed, with offset
              "line_end": end_idx + 1 + body_offset,
          })
      return records


  def command_review_extract(args: argparse.Namespace) -> int:
      runs_root = FACTORY_RUNS_ROOT
      slug_dir = runs_root / args.slug
      reviews_dir = slug_dir / "reviews"
      if not reviews_dir.exists():
          print(f"reviews dir not found: {reviews_dir}", file=sys.stderr)
          return 2

      pattern = f"{args.stage}.*.review.md"
      files = sorted(reviews_dir.glob(pattern))

      output_records = []
      for fp in files:
          text = fp.read_text(encoding="utf-8")
          try:
              fm, body_offset = _parse_frontmatter(text)
          except ValueError as exc:
              print(f"error: malformed frontmatter in {fp}: {exc}", file=sys.stderr)
              return 2
          body_lines = text.splitlines()[body_offset:]
          records = _extract_findings(body_lines, body_offset)
          for rec in records:
              rec["review"] = str(fp.relative_to(REPO_ROOT)) if fp.is_relative_to(REPO_ROOT) else str(fp)
              rec["frontmatter_status"] = fm.get("resolution_status") or None
              output_records.append(rec)

      if args.format == "json":
          out_text = json.dumps(output_records, indent=2)
      else:  # jsonl
          out_text = "\n".join(json.dumps(r) for r in output_records)

      if args.out:
          Path(args.out).write_text(out_text)
      else:
          print(out_text)
      return 0
  ```
