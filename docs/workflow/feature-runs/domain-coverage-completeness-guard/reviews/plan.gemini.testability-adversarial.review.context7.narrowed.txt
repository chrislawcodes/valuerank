00035:  * Note: Progress is only updated for runs in PENDING or RUNNING state.

00048:     select: { progress: true, status: true },
00049:   });
00050:

00052:     throw new NotFoundError('Run', runId);
00053:   }

00055:   // Don't update progress if run is not in PENDING or RUNNING state
00056:   // This prevents overcounting when probe jobs complete after the run has moved on
